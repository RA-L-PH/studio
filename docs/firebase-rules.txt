# PulseQueue Firebase Security Rules (RTDB Centric Version)

## Realtime Database Rules
# APPLY THESE IN THE 'Rules' TAB OF THE REALTIME DATABASE SECTION IN THE FIREBASE CONSOLE.
# Location: https://console.firebase.google.com/project/_/database/rules

{
  "rules": {
    "queues": {
      "$clinicCode": {
        // Allow read/write access if the user's UID is associated with the clinic code
        ".read": "root.child('clinic_owners').child(auth.uid).val() == $clinicCode",
        ".write": "root.child('clinic_owners').child(auth.uid).val() == $clinicCode",
        ".indexOn": ["token_number", "status"]
      }
    },
    "clinic_owners": {
      "$uid": {
        // Allow a user to write their own clinic code mapping
        ".write": "auth != null && auth.uid == $uid"
      }
    }
  }
}



## Firestore Rules
# Location: firestore.rules (Keep these locked as we have migrated to RTDB)
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow users to create and manage their own clinic profile
    match /clinics/{userId} {
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.resource.data.clinicCode != null;
    }
    
    // Ensure clinicCode is unique
    match /clinic_codes/{clinicCode} {
    	allow read: if true;
      allow create: if request.auth != null;
    }

    // Fallback for other collections, requires authentication
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}

## Implementation Notes:
1. These RTDB rules are permissive (read/write: true) to ensure zero permission errors during your MVP development.
2. In the Firebase Console, ensure you have initialized your Realtime Database in the same region as your project.
3. The 'queues' indexing is added for better performance when the list grows.
4. For production, replace global access with 'auth != null' or specific user-level checks.
